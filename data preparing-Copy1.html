def load_images_from_folder(folder):
    images = []
    for filename in os.listdir(folder):
        img = cv2.imread(os.path.join(folder,filename),cv2.IMREAD_GRAYSCALE)
        img = cv2.resize(img,fixed_size)
        if img is not None:
            images.append(img)
    return images
